#include <iostream>
#include <stdlib.h>
#define bktsize 400
using namespace std;

void bktinput(int ps,int op){
    if(ps>bktsize)
        cout<<"Bucket Overflow"<<endl;
    else{
        while(ps>op){
            cout<<op<<" bytes outputted"<<endl;
            ps-=op;
        }
        if(ps>0)
            cout<<ps<<" bytes sent"<<endl;
    }
}

int main(){
    int ps,op,np;
    cout<<"Enter output Rate : ";
    cin>>op;
    cout<<"\nEnter no. of packets : ";
    cin>>np;
    for(int i = 1; i<=np;i++){
        cout<<"Enter size of packet "<<i<<" : ";
        cin>>ps;
        bktinput(ps,op);
    }
}
